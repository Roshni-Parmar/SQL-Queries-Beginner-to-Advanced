CREATE TABLE customer(
	cust_id NUMBER(5) PRIMARY KEY,
	cust_name VARCHAR2(20) NOT NULL,
	gender CHAR(1) CHECK(gender IN('M','F')),
	dob DATE,
	phone NUMBER(15) UNIQUE,
	email VARCHAR2(20) UNIQUE,
	address VARCHAR2(25),
	pan VARCHAR2(10) UNIQUE,
	aadhar  VARCHAR2(12) UNIQUE
);

CREATE TABLE account(
	acno NUMBER PRIMARY KEY,
	cust_id NUMBER(5) NOT NULL,
	acType VARCHAR2(10) CHECK(acType IN('saving','current')),
	bal NUMBER CHECK(bal >=0),
	opening DATE NOT NULL,
	status VARCHAR2(10) CHECK(status IN('active','closed','blocked')),
	CONSTRAINT fk_ac_cust
	FOREIGN KEY (cust_id)
	REFERENCES customer(cust_id)
);

CREATE TABLE transcation( 
	tran_id NUMBER PRIMARY KEY,
	acno NUMBER,
	tran_date DATE,
	tran_type VARCHAR2(15) CHECK(tran_type IN('credit','debit')),
	amt NUMBER CHECK(amt>0),
	payment_mode VARCHAR2(10) CHECK(payment_mode IN('online','cash','cheque')),
	CONSTRAINT fk_tran_ac
	FOREIGN KEY (acno)
	REFERENCES account(acno)
);

CREATE TABLE loan(
	loan_id NUMBER PRIMARY KEY,
	cust_id NUMBER,
	loan_type VARCHAR2(20) CHECK (loan_type IN('home','car','education','personal')),
	loan_amt NUMBER CHECK(loan_amt>1000),
	int_rate NUMBER CHECK(int_rate BETWEEN 1 AND 20),
	loan_date DATE,
	loan_status VARCHAR2(20) CHECK(loan_status IN('approved','pending','rejected')), 
	CONSTRAINT fk_load_customer
	FOREIGN KEY (cust_id)
	REFERENCES customer(cust_id)
);

CREATE TABLE branch(
	branch_id NUMBER(3) PRIMARY KEY,
	branch_name VARCHAR2(20) NOT NULL,
	branch_code  VARCHAR2(10) UNIQUE,
	phone NUMBER(15) UNIQUE
);

CREATE TABLE doctor(
	doctor_id NUMBER PRIMARY KEY,
	doctor_name VARCHAR2(10) NOT NULL,
	gender CHAR(1) CHECK(gender IN('M','F')),
	specialization VARCHAR2(20) NOT NULL,
	qualification VARCHAR2(20),
	experience NUMBER(2) CHECK(experience >=0), 
	phone VARCHAR2(10) UNIQUE,
	email VARCHAR2(20) UNIQUE,
	sal NUMBER(10,2) CHECK(sal>0),
	dep VARCHAR2(10) NOT NULL,
	joining_date DATE
);


CREATE TABLE patient(
	patient_id NUMBER PRIMARY KEY,
	patient_name VARCHAR2(20) NOT NULL,
	gender CHAR(1) CHECK(gender IN('M','F')),
	dob DATE,
	age NUMBER(2) CHECK(age BETWEEN 1 AND 100),
	blood_group CHAR(3) CHECK(blood_group IN('A+','A-','O-','O+','B+','B-','AB+','AB-')),
	phone VARCHAR2(10) UNIQUE,
	email VARCHAR2(20) UNIQUE,
	address VARCHAR2(20),
	doctor_id NUMBER,
	
	CONSTRAINT fk_patient_doctor
	FOREIGN KEY (doctor_id)
	REFERENCES doctor(doctor_id)
);

CREATE TABLE appointment(
	app_id NUMBER PRIMARY KEY,
	patient_id NUMBER NOT NULL,
	doctor_id NUMBER NOT NULL,
	app_date DATE NOT NULL,
	app_time TIMESTAMP NOT NULL,
	app_status VARCHAR2(10) CHECK(app_status IN('scheduled','completed','cancelled')),
	consulation_fee NUMBER CHECK(consulation_fee>=0),
	
	CONSTRAINT fk_app_patient
	FOREIGN KEY (patient_id)
	REFERENCES patient(patient_id),
	
	CONSTRAINT fk_app_doctor
	FOREIGN KEY (doctor_id)
	REFERENCES doctor(doctor_id)
	
);

CREATE TABLE medicine(
	med_id NUMBER PRIMARY KEY,
	med_name VARCHAR2(20) NOT NULL,
	med_type VARCHAR2(20) CHECK(med_type in('tablet','syrup','injection','capsule')),
	manuf VARCHAR2(20),
	price NUMBER CHECK(price>0),
	stock_qty NUMBER CHECK(stock_qty>=0),
	expiry_date DATE NOT NULL
);

CREATE TABLE billing(
	bill_id NUMBER PRIMARY KEY,
	patient_id NUMBER,
	app_id NUMBER,
	bill_date DATE NOT NULL,
	consulation_charge NUMBER CHECK(consulation_chare>=0),
	med_charge NUMBER CHECK(med_charge>=0),
	total_amt NUMBER CHECK(total_amt>=0),
	pay_method VARCHAR2(10) CHECK(pay_method IN('cash','online','card','upi')),
	pay_status VARCHAR2(10) CHECK(pay_status in('paid','pending')),
	
	CONSTRAINT fk_billing_patient
	FOREIGN KEY (patient_id)
	REFERENCES patient(patient_id),
	
	CONSTRAINT fk_billing_appointment
	FOREIGN KEY (app_id)
	REFERENCES appointment(app_id)
);

Relationships to Apply
Doctor (1) → (Many) Patient
Doctor ID in Patient is a Foreign Key.
Doctor (1) → (Many) Appointment
Doctor ID in Appointment is a Foreign Key.
Patient (1) → (Many) Appointment
Patient ID in Appointment is a Foreign Key.
Patient (1) → (Many) Billing
Patient ID in Billing is a Foreign Key.
Appointment (1) → (1 or Many) Billing
Appointment ID in Billing is a Foreign Key.

CREATE TABLE customer(
	cust_id NUMBER PRIMARY KEY,
	fn VARCHAR2(10) NOT NULL,
	ln VARCHAR2(10) NOT NULL,
	gender CHAR(1) CHECK(gender IN ('M','F')),
	dob DATE,
	email VARCHAR2(20) UNIQUE,
	phone VARCHAR2(10) UNIQUE,
	address VARCHAR2(20),
	city VARCHAR2(10),
	state VARCHAR2(10),
	pin_code NUMBER(6) CHECK(pin_code BETWEEN 100000 AND 999999),
	registration_date DATE NOT NULL
);

CREATE TABLE orders(
	order_id NUMBER PRIMARY KEY,
	cust_id NUMBER NOT NULL,
	order_date DATE NOT NULL,
	order_status VARCHAR2(10)  CHECK(order_status in('pending','confirmed','packed','shipped','delivered','cancelled')),
	pay_method VARCHAR2(10) CHECK(pay_method in('cash','card','upi','netbanking')),
	shipping_add VARCHAR2(10) NOT NULL,
	total_amt NUMBER CHECK(total_amt>=0),
	
	CONSTRAINT fk_order_cust
	FOREIGN KEY (cust_id)
	REFERENCES customer(cust_id)
	
);

CREATE TABLE product(
	prod_id NUMBER PRIMARY KEY,
	prod_name VARCHAR2(10) NOT NULL,
	prod_price NUMBER CHECK(prod_price>=0),
	stock_qty NUMBER CHECK(stock_qty>=0),
	category_id NUMBER,
	supplier_id NUMBER,
	
	CONSTRAINT fk_prod_category 
	FOREIGN KEY (category_id)
	REFERENCES category(category_id),
	
	CONSTRAINT fk_prod_supplier 
	FOREIGN KEY (supplier_id)
	REFERENCES supplier(supplier_id)
	
);

CREATE TABLE category(
	category_id NUMBER PRIMARY KEY,
	category_name VARCHAR2(10) NOT NULL,
	category_descrip VARCHAR2(10)
);

CREATE TABLE supplier(
	supplier_id NUMBER PRIMARY KEY,
	supplier_name VARCHAR2(10) NOT NULL,
	contact VARCHAR2(10) UNIQUE,
	email VARCHAR2(20) UNIQUE,
	add VARCHAR2(20),
	city VARCHAR2(10)
);

CREATE TABLE order_item(
	ord_item_id NUMBER PRIMARY KEY,
	order_id NUMBER,
	prod_id NUMBER,
	qty NUMBER CHECK(qty>0),
	unit_price NUMBER CHECK(unit_price>0),
	disc NUMBER CHECK(disc BETWEEN 0 AND 100),
	subtotal NUMBER CHECK(subtotal>0),
	
	CONSTRAINT fk_order_item_order
	FOREIGN KEY (order_id)
	REFERENCES orders(order_id),
	
	CONSTRAINT fk_orderitem_product
	FOREIGN KEY (prod_id)
	REFERENCES product(prod_id)
);
